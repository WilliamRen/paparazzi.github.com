60 /* Big Endian to Float */
61 static inline float bef(volatile uint8_t *c) {
62  float f;
63  int8_t * p;
64  p = ((int8_t *)&f)+3;
65  *p-- = *c++;
66  *p-- = *c++;
67  *p-- = *c++;
68  *p = *c;
69  return f;
70 }

72 static inline bool_t gx3_verify_chk(volatile uint8_t *buff_add) {
73  uint16_t i,chk_calc;
74  chk_calc = 0;
75  for (i=0;i<GX3_MSG_LEN-2;i++) {
76  chk_calc += (uint8_t)*buff_add++;
77  }
78  return (chk_calc == ( (((uint16_t)*buff_add)<<8) + (uint8_t)*(buff_add+1) ));
79 }

294 /* GX3 Packet Collection */
295 void gx3_packet_parse( uint8_t c ) {
296  switch (ahrs_impl.gx3_packet.status) {
297  case GX3PacketWaiting:
298  ahrs_impl.gx3_packet.msg_idx = 0;
299  if (c == GX3_HEADER) {
300  ahrs_impl.gx3_packet.status++;
301  ahrs_impl.gx3_packet.msg_buf[ahrs_impl.gx3_packet.msg_idx] = c;
302  ahrs_impl.gx3_packet.msg_idx++;
303  } else {
304  ahrs_impl.gx3_packet.hdr_error++;
305  }
306  break;
307  case GX3PacketReading:
308  ahrs_impl.gx3_packet.msg_buf[ahrs_impl.gx3_packet.msg_idx] = c;
309  ahrs_impl.gx3_packet.msg_idx++;
310  if (ahrs_impl.gx3_packet.msg_idx == GX3_MSG_LEN) {
311  if (gx3_verify_chk(ahrs_impl.gx3_packet.msg_buf)) {
312  ahrs_impl.gx3_packet.msg_available = TRUE;
313  } else {
314  ahrs_impl.gx3_packet.msg_available = FALSE;
315  ahrs_impl.gx3_packet.chksm_error++;
316  }
317  ahrs_impl.gx3_packet.status = 0;
318  }
319  break;
320  default:
321  ahrs_impl.gx3_packet.status = 0;
322  ahrs_impl.gx3_packet.msg_idx = 0;
323  break;
324  }
325 }
