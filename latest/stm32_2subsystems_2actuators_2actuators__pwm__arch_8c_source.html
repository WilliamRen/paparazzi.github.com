104 static inline void actuators_pwm_arch_channel_init(uint32_t timer_peripheral,
105  enum tim_oc_id oc_id) {
106 
107  timer_disable_oc_output(timer_peripheral, oc_id);
108  //There is no such register in TIM9 and 12.
109  if (timer_peripheral != TIM9 && timer_peripheral != TIM12)
110  timer_disable_oc_clear(timer_peripheral, oc_id);
111  timer_enable_oc_preload(timer_peripheral, oc_id);
112  timer_set_oc_slow_mode(timer_peripheral, oc_id);
113  timer_set_oc_mode(timer_peripheral, oc_id, TIM_OCM_PWM1);
114  timer_set_oc_polarity_high(timer_peripheral, oc_id);
115  timer_enable_oc_output(timer_peripheral, oc_id);
116  // Used for TIM1 and TIM8, the function does nothing if other timer is specified.
117  timer_enable_break_main_output(timer_peripheral);
118 }

138 static inline void set_servo_timer(uint32_t timer, uint32_t period, uint8_t channels_mask) {
139  timer_reset(timer);
140 
141  /* Timer global mode:
142  * - No divider.
143  * - Alignement edge.
144  * - Direction up.
145  */
146  if ((timer == TIM9) || (timer == TIM12))
147  //There are no EDGE and DIR settings in TIM9 and TIM12
148  timer_set_mode(timer, TIM_CR1_CKD_CK_INT, 0, 0);
149  else
150  timer_set_mode(timer, TIM_CR1_CKD_CK_INT, TIM_CR1_CMS_EDGE, TIM_CR1_DIR_UP);
151 
152 
153  // TIM1, 8 and 9 use APB2 clock, all others APB1
154  if (timer != TIM1 && timer != TIM8 && timer != TIM9) {
155  timer_set_prescaler(timer, (TIMER_APB1_CLK / ONE_MHZ_CLK) - 1); // 1uS
156  } else {
157  // TIM9, 1 and 8 use APB2 clock
158  timer_set_prescaler(timer, (TIMER_APB2_CLK / ONE_MHZ_CLK) - 1);
159  }
160 
161  timer_disable_preload(timer);
162 
163  timer_continuous_mode(timer);
164 
165  timer_set_period(timer, (ONE_MHZ_CLK / period) - 1);
166 
167  /* Disable outputs and configure channel if needed. */
168  if (bit_is_set(channels_mask, 0)) {
169  actuators_pwm_arch_channel_init(timer, TIM_OC1);
170  }
171  if (bit_is_set(channels_mask, 1)) {
172  actuators_pwm_arch_channel_init(timer, TIM_OC2);
173  }
174  if (bit_is_set(channels_mask, 2)) {
175  actuators_pwm_arch_channel_init(timer, TIM_OC3);
176  }
177  if (bit_is_set(channels_mask, 3)) {
178  actuators_pwm_arch_channel_init(timer, TIM_OC4);
179  }
180 
181  /*
182  * Set initial output compare values.
183  * Note: Maybe we should preload the compare registers with some sensible
184  * values before we enable the timer?
185  */
186  //timer_set_oc_value(timer, TIM_OC1, 1000);
187  //timer_set_oc_value(timer, TIM_OC2, 1000);
188  //timer_set_oc_value(timer, TIM_OC3, 1000);
189  //timer_set_oc_value(timer, TIM_OC4, 1000);
190 
191  /* -- Enable timer -- */
192  /*
193  * ARR reload enable.
194  * Note: In our case it does not matter much if we do preload or not. As it
195  * is unlikely we will want to change the frequency of the timer during
196  * runtime anyways.
197  */
198  timer_enable_preload(timer);
199 
200  /* Counter enable. */
201  timer_enable_counter(timer);
202 
203 }
