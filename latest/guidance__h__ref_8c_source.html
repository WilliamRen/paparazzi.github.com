91 void gh_set_ref(struct Int32Vect2 pos, struct Int32Vect2 speed, struct Int32Vect2 accel) {
92  struct Int64Vect2 new_pos;
93  new_pos.x = ((int64_t)pos.x)<<(GH_POS_REF_FRAC - INT32_POS_FRAC);
94  new_pos.y = ((int64_t)pos.y)<<(GH_POS_REF_FRAC - INT32_POS_FRAC);

100 void gh_update_ref_from_pos_sp(struct Int32Vect2 pos_sp) {
101 
102  VECT2_ADD(gh_pos_ref, gh_speed_ref);
103  VECT2_ADD(gh_speed_ref, gh_accel_ref);
104 
105  // compute the "speed part" of accel = -2*zeta*omega*speed -omega^2(pos - pos_sp)
106  struct Int32Vect2 speed;
107  INT32_VECT2_RSHIFT(speed, gh_speed_ref, (GH_SPEED_REF_FRAC - GH_ACCEL_REF_FRAC));
108  VECT2_SMUL(speed, speed, -2 * gh_zeta_omega);
109  INT32_VECT2_RSHIFT(speed, speed, GH_ZETA_OMEGA_FRAC);
110  // compute pos error in pos_sp resolution
111  struct Int32Vect2 pos_err;
112  INT32_VECT2_RSHIFT(pos_err, gh_pos_ref, (GH_POS_REF_FRAC - INT32_POS_FRAC));
113  VECT2_DIFF(pos_err, pos_err, pos_sp);
114  // convert to accel resolution
115  INT32_VECT2_RSHIFT(pos_err, pos_err, (INT32_POS_FRAC - GH_ACCEL_REF_FRAC));
116  // compute the "pos part" of accel
117  struct Int32Vect2 pos;
118  VECT2_SMUL(pos, pos_err, -gh_omega_2);
119  INT32_VECT2_RSHIFT(pos, pos, GH_OMEGA_2_FRAC);
120  // sum accel
121  VECT2_SUM(gh_accel_ref, speed, pos);
122 
123  /* Compute max ref accel/speed along route before saturation */
124  gh_compute_ref_max(&pos_err);
125 
126  gh_saturate_ref_accel();
127  gh_saturate_ref_speed();
128 }

154 static void gh_compute_route_ref(struct Int32Vect2* ref_vector) {
155  float f_route_ref = atan2f(-ref_vector->y, -ref_vector->x);
156  route_ref = ANGLE_BFP_OF_REAL(f_route_ref);
157  /* Compute North and East route components */
158  PPRZ_ITRIG_SIN(s_route_ref, route_ref);
159  PPRZ_ITRIG_COS(c_route_ref, route_ref);
160  c_route_ref = abs(c_route_ref);
161  s_route_ref = abs(s_route_ref);
162 }

164 static void gh_compute_ref_max(struct Int32Vect2* ref_vector) {
165  /* Compute route reference before saturation */
166  if (ref_vector->x == 0 && ref_vector->y == 0) {
167  gh_max_accel_ref.x = 0;
168  gh_max_accel_ref.y = 0;
169  gh_max_speed_ref.x = 0;
170  gh_max_speed_ref.y = 0;
171  }
172  else {
173  gh_compute_route_ref(ref_vector);
174  /* Compute maximum acceleration*/
175  gh_max_accel_ref.x = INT_MULT_RSHIFT(gh_max_accel, c_route_ref, INT32_TRIG_FRAC);
176  gh_max_accel_ref.y = INT_MULT_RSHIFT(gh_max_accel, s_route_ref, INT32_TRIG_FRAC);
177  /* Compute maximum speed*/
178  gh_max_speed_ref.x = INT_MULT_RSHIFT(gh_max_speed, c_route_ref, INT32_TRIG_FRAC);
179  gh_max_speed_ref.y = INT_MULT_RSHIFT(gh_max_speed, s_route_ref, INT32_TRIG_FRAC);
180  /* restore gh_speed_ref range (Q14.17) */
181  INT32_VECT2_LSHIFT(gh_max_speed_ref, gh_max_speed_ref, (GH_SPEED_REF_FRAC - GH_MAX_SPEED_REF_FRAC));
182  }
183 }

216 static void gh_saturate_ref_accel(void) {
217  /* Saturate accelerations */
218  if (gh_accel_ref.x <= -gh_max_accel_ref.x) {
219  gh_accel_ref.x = -gh_max_accel_ref.x;
220  }
221  else if (gh_accel_ref.x >= gh_max_accel_ref.x) {
222  gh_accel_ref.x = gh_max_accel_ref.x;
223  }
224  if (gh_accel_ref.y <= -gh_max_accel_ref.y) {
225  gh_accel_ref.y = -gh_max_accel_ref.y;
226  }
227  else if (gh_accel_ref.y >= gh_max_accel_ref.y) {
228  gh_accel_ref.y = gh_max_accel_ref.y;
229  }
