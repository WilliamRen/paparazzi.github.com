158 void imu_aspirin2_event(void)
159 {
160  mpu60x0_spi_event(&imu_aspirin2.mpu);
161  if (imu_aspirin2.mpu.data_available) {
162  /* HMC5883 has xzy order of axes in returned data */
163  struct Int32Vect3 mag;
164  mag.x = Int16FromBuf(imu_aspirin2.mpu.data_ext, 0);
165  mag.z = Int16FromBuf(imu_aspirin2.mpu.data_ext, 2);
166  mag.y = Int16FromBuf(imu_aspirin2.mpu.data_ext, 4);
167 #ifdef LISA_M_LONGITUDINAL_X
168  RATES_ASSIGN(imu.gyro_unscaled,
169  imu_aspirin2.mpu.data_rates.rates.q,
170  -imu_aspirin2.mpu.data_rates.rates.p,
171  imu_aspirin2.mpu.data_rates.rates.r);
172  VECT3_ASSIGN(imu.accel_unscaled,
173  imu_aspirin2.mpu.data_accel.vect.y,
174  -imu_aspirin2.mpu.data_accel.vect.x,
175  imu_aspirin2.mpu.data_accel.vect.z);
176  VECT3_ASSIGN(imu.mag_unscaled, -mag.x, -mag.y, mag.z);
177 #else
178 #ifdef LISA_S
179 #ifdef LISA_S_UPSIDE_DOWN
180  RATES_ASSIGN(imu.gyro_unscaled,
181  imu_aspirin2.mpu.data_rates.rates.p,
182  -imu_aspirin2.mpu.data_rates.rates.q,
183  -imu_aspirin2.mpu.data_rates.rates.r);
184  VECT3_ASSIGN(imu.accel_unscaled,
185  imu_aspirin2.mpu.data_accel.vect.x,
186  -imu_aspirin2.mpu.data_accel.vect.y,
187  -imu_aspirin2.mpu.data_accel.vect.z);
188  VECT3_ASSIGN(imu.mag_unscaled, mag.x, -mag.y, -mag.z);
189 #else
190  RATES_COPY(imu.gyro_unscaled, imu_aspirin2.mpu.data_rates.rates);
191  VECT3_COPY(imu.accel_unscaled, imu_aspirin2.mpu.data_accel.vect);
192  VECT3_COPY(imu.mag_unscaled, mag);
193 #endif
194 #else
195  RATES_COPY(imu.gyro_unscaled, imu_aspirin2.mpu.data_rates.rates);
196  VECT3_COPY(imu.accel_unscaled, imu_aspirin2.mpu.data_accel.vect);
197  VECT3_ASSIGN(imu.mag_unscaled, mag.y, -mag.x, mag.z)
198 #endif
199 #endif
200  imu_aspirin2.mpu.data_available = FALSE;
201  imu_aspirin2.gyro_valid = TRUE;
202  imu_aspirin2.accel_valid = TRUE;
203  imu_aspirin2.mag_valid = TRUE;
204  }
205 }

217 bool_t imu_aspirin2_configure_mag_slave(Mpu60x0ConfigSet mpu_set, void* mpu)
218 {
219  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_ADDR, (HMC58XX_ADDR >> 1));
220  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_REG, HMC58XX_REG_CFGA);
221  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_DO, HMC58XX_CRA);
222  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_CTRL, (1 << 7)); // Slave 4 enable
223 
224  mpu_wait_slave4_ready();
225 
226  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_ADDR, (HMC58XX_ADDR >> 1));
227  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_REG, HMC58XX_REG_CFGB);
228  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_DO, HMC58XX_CRB);
229  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_CTRL, (1 << 7)); // Slave 4 enable
230 
231  mpu_wait_slave4_ready();
232 
233  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_ADDR, (HMC58XX_ADDR >> 1));
234  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_REG, HMC58XX_REG_MODE);
235  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_DO, HMC58XX_MD);
236  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_CTRL, (1 << 7)); // Slave 4 enable
237 
238  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV0_ADDR, (HMC58XX_ADDR >> 1) | MPU60X0_SPI_READ);
239  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV0_REG, HMC58XX_REG_DATXM);
240  // Put the enable command as last.
241  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV0_CTRL,
242  (1 << 7) | // Slave 0 enable
243  (6 << 0) ); // Read 6 bytes
244 
245  return TRUE;
246 }

248 void mpu_wait_slave4_ready(void)
249 {
250  while (!imu_aspirin2.slave4_ready) {
251  if (imu_aspirin2.wait_slave4_trans.status == SPITransDone) {
252  imu_aspirin2.wait_slave4_tx_buf[0] = MPU60X0_REG_I2C_MST_STATUS | MPU60X0_SPI_READ;
253  spi_submit(imu_aspirin2.mpu.spi_p, &(imu_aspirin2.wait_slave4_trans));
254  }
255  }
256 }

258 void mpu_wait_slave4_ready_cb(struct spi_transaction * t)
259 {
260  if (bit_is_set(t->input_buf[1], MPU60X0_I2C_SLV4_DONE))
261  imu_aspirin2.slave4_ready = TRUE;
262  else
263  imu_aspirin2.slave4_ready = FALSE;
264  t->status = SPITransDone;
