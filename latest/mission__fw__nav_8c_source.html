79 static inline bool_t mission_nav_path(struct _mission_path * path) {
80  if (path->nb == 0) {
81  return FALSE; // nothing to do
82  }
83  if (path->nb == 1) {
84  // handle as a single waypoint
85  struct _mission_wp wp = { path->path[0] };
86  return mission_nav_wp(&wp);
87  }
88  if (path->path_idx == path->nb-1) {
89  last_wp = path->path[path->path_idx]; // store last wp
90  return FALSE; // end of path
91  }
92  // normal case
93  struct EnuCoor_f from = path->path[path->path_idx];
94  struct EnuCoor_f to = path->path[path->path_idx+1];
95  nav_route_xy(from.x, from.y, to.x, to.y);
96  NavVerticalAutoThrottleMode(0.);
97  NavVerticalAltitudeMode(to.z, 0.); // both altitude should be the same anyway
98  if (nav_approaching_xy(to.x, to.y, from.x, from.y, CARROT)) {
99  path->path_idx++; // go to next segment
100  }
101  return TRUE;
102 }

105 int mission_run() {
106  // current element
107  struct _mission_element * el = NULL;
108  if ((el = mission_get()) == NULL) {
109  // TODO do something special like a waiting circle before ending the mission ?
110  return FALSE; // end of mission
111  }
112 
113  bool_t el_running = FALSE;
114  switch (el->type){
115  case MissionWP:
116  el_running = mission_nav_wp(&(el->element.mission_wp));
117  break;
118  case MissionCircle:
119  el_running = mission_nav_circle(&(el->element.mission_circle));
120  break;
121  case MissionSegment:
122  el_running = mission_nav_segment(&(el->element.mission_segment));
123  break;
124  case MissionPath:
125  el_running = mission_nav_path(&(el->element.mission_path));
126  break;
127  default:
128  // invalid type or pattern not yet handled
129  break;
130  }
131 
132  // increment element_time
133  mission.element_time += dt_navigation;
134 
135  // test if element is finished or element time is elapsed
136  if (!el_running || (el->duration > 0. && mission.element_time >= el->duration)) {
137  // reset timer
138  mission.element_time = 0.;
139  // go to next element
140  mission.current_idx++;
141  }
142 
143  return TRUE;
144 }
