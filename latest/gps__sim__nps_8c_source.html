34 void gps_feed_value() {
35  gps.ecef_pos.x = sensors.gps.ecef_pos.x * 100.;
36  gps.ecef_pos.y = sensors.gps.ecef_pos.y * 100.;
37  gps.ecef_pos.z = sensors.gps.ecef_pos.z * 100.;
38  gps.ecef_vel.x = sensors.gps.ecef_vel.x * 100.;
39  gps.ecef_vel.y = sensors.gps.ecef_vel.y * 100.;
40  gps.ecef_vel.z = sensors.gps.ecef_vel.z * 100.;
41  //ecef pos seems to be based on geocentric model, hence we get a very high alt when converted to lla
42  gps.lla_pos.lat = sensors.gps.lla_pos.lat * 1e7;
43  gps.lla_pos.lon = sensors.gps.lla_pos.lon * 1e7;
44  gps.lla_pos.alt = sensors.gps.lla_pos.alt * 1000.;
45  gps.hmsl = sensors.gps.hmsl * 1000.;
46 
47  /* calc NED speed from ECEF */
48  struct LtpDef_d ref_ltp;
49  ltp_def_from_ecef_d(&ref_ltp, &sensors.gps.ecef_pos);
50  struct NedCoor_d ned_vel_d;
51  ned_of_ecef_vect_d(&ned_vel_d, &ref_ltp, &sensors.gps.ecef_vel);
52  gps.ned_vel.x = ned_vel_d.x * 100;
53  gps.ned_vel.y = ned_vel_d.y * 100;
54  gps.ned_vel.z = ned_vel_d.z * 100;
55 
56  /* horizontal and 3d ground speed in cm/s */
57  gps.gspeed = sqrt(ned_vel_d.x * ned_vel_d.x + ned_vel_d.y * ned_vel_d.y) * 100;
58  gps.speed_3d = sqrt(ned_vel_d.x * ned_vel_d.x + ned_vel_d.y * ned_vel_d.y + ned_vel_d.z * ned_vel_d.z) * 100;
59 
60  /* ground course in radians * 1e7 */
61  gps.course = atan2(ned_vel_d.y, ned_vel_d.x) * 1e7;
62 
63 #if GPS_USE_LATLONG
64  /* Computes from (lat, long) in the referenced UTM zone */
65  struct LlaCoor_f lla_f;
66  lla_f.lat = ((float) gps.lla_pos.lat) / 1e7;
67  lla_f.lon = ((float) gps.lla_pos.lon) / 1e7;
68  struct UtmCoor_f utm_f;
69  utm_f.zone = nav_utm_zone0;
70  /* convert to utm */
71  utm_of_lla_f(&utm_f, &lla_f);
72  /* copy results of utm conversion */
73  gps.utm_pos.east = utm_f.east*100;
74  gps.utm_pos.north = utm_f.north*100;
75  gps.utm_pos.alt = gps.lla_pos.alt;
76  gps.utm_pos.zone = nav_utm_zone0;
77 #endif
78 
79  if (gps_has_fix)
80  gps.fix = GPS_FIX_3D;
81  else
82  gps.fix = GPS_FIX_NONE;
83  gps_available = TRUE;
84 }
