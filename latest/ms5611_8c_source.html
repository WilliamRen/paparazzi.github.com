36 bool_t ms5611_prom_crc_ok(uint16_t* prom) {
37  int32_t i, j;
38  uint32_t res = 0;
39  uint8_t crc = prom[7] & 0xF;
40  prom[7] &= 0xFF00;
41  for (i = 0; i < 16; i++) {
42  if (i & 1)
43  res ^= ((prom[i>>1]) & 0x00FF);
44  else
45  res ^= (prom[i>>1]>>8);
46  for (j = 8; j > 0; j--) {
47  if (res & 0x8000)
48  res ^= 0x1800;
49  res <<= 1;
50  }
51  }
52  prom[7] |= crc;
53  if (crc == ((res >> 12) & 0xF))
54  return TRUE;
55  else
56  return FALSE;
57 }

62 void ms5611_calc(struct Ms5611Data* ms) {
63  int64_t dt, tempms, off, sens, t2, off2, sens2;
64 
65  /* difference between actual and ref temperature */
66  dt = ms->d2 - (int64_t)ms->c[5] * (1<<8);
67  /* actual temperature */
68  tempms = 2000 + ((int64_t)dt * ms->c[6]) / (1<<23);
69  /* offset at actual temperature */
70  off = ((int64_t)ms->c[2] * (1<<16)) + ((int64_t)ms->c[4] * dt) / (1<<7);
71  /* sensitivity at actual temperature */
72  sens = ((int64_t)ms->c[1] * (1<<15)) + ((int64_t)ms->c[3] * dt) / (1<<8);
73  /* second order temperature compensation */
74  if (tempms < 2000) {
75  t2 = (dt*dt) / (1<<31);
76  off2 = 5 * ((int64_t)(tempms-2000)*(tempms-2000)) / (1<<1);
77  sens2 = 5 * ((int64_t)(tempms-2000)*(tempms-2000)) / (1<<2);
78  if (tempms < -1500) {
79  off2 = off2 + 7 * (int64_t)(tempms+1500)*(tempms+1500);
80  sens2 = sens2 + 11 * ((int64_t)(tempms+1500)*(tempms+1500)) / (1<<1);
81  }
82  tempms = tempms - t2;
83  off = off - off2;
84  sens = sens - sens2;
85  }
86 
87  /* temperature in deg Celsius with 0.01 degC resolultion */
88  ms->temperature = (int32_t)tempms;
89  /* temperature compensated pressure in Pascal (0.01mbar) */
90  ms->pressure = (uint32_t)((((int64_t)ms->d1 * sens) / (1<<21) - off) / (1<<15));
91 }
