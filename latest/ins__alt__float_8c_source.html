103 #if USE_BAROMETER
104 static void baro_cb(uint8_t __attribute__((unused)) sender_id, const float *pressure) {
105  if (!ins_baro_initialized) {
106  ins_qfe = *pressure;
107  ins_baro_initialized = TRUE;
108  }
109  if (ins.vf_realign) {
110  ins.vf_realign = FALSE;
111  ins_alt = ground_alt;
112  ins_alt_dot = 0.;
113  ins_qfe = *pressure;
114  alt_kalman_reset();
115  }
116  else { /* not realigning, so normal update with baro measurement */
117  ins_baro_alt = ground_alt + pprz_isa_height_of_pressure(*pressure, ins_qfe);
118  /* run the filter */
119  EstimatorSetAlt(ins_baro_alt);
120  /* set new altitude, just copy old horizontal position */
121  struct UtmCoor_f utm;
122  UTM_COPY(utm, *stateGetPositionUtm_f());
123  utm.alt = ins_alt;
124  stateSetPositionUtm_f(&utm);
125  struct NedCoor_f ned_vel;
126  memcpy(&ned_vel, stateGetSpeedNed_f(), sizeof(struct NedCoor_f));
127  ned_vel.z = -ins_alt_dot;
128  stateSetSpeedNed_f(&ned_vel);
129  }
130 }
131 #endif

194 void alt_kalman(float z_meas) {
195  float DT;
196  float R;
197  float SIGMA2;
198 
199 #if USE_BAROMETER
200 #ifdef SITL
201  DT = BARO_SIM_DT;
202  R = 0.5;
203  SIGMA2 = 0.1;
204 #elif USE_BARO_MS5534A
205  if (alt_baro_enabled) {
206  DT = BARO_DT;
207  R = baro_MS5534A_r;
208  SIGMA2 = baro_MS5534A_sigma2;
209  } else
210 #elif USE_BARO_ETS
211  if (baro_ets_enabled) {
212  DT = BARO_ETS_DT;
213  R = baro_ets_r;
214  SIGMA2 = baro_ets_sigma2;
215  } else
216 #elif USE_BARO_MS5611
217  if (baro_ms5611_enabled) {
218  DT = BARO_MS5611_DT;
219  R = baro_ms5611_r;
220  SIGMA2 = baro_ms5611_sigma2;
221  } else
222 #elif USE_BARO_AMSYS
223  if (baro_amsys_enabled) {
224  DT = BARO_AMSYS_DT;
225  R = baro_amsys_r;
226  SIGMA2 = baro_amsys_sigma2;
227  } else
228 #elif USE_BARO_BMP
229  if (baro_bmp_enabled) {
230  DT = BARO_BMP_DT;
231  R = baro_bmp_r;
232  SIGMA2 = baro_bmp_sigma2;
233  } else
234 #endif
235 #endif // USE_BAROMETER
236  {
237  DT = GPS_DT;
238  R = GPS_R;
239  SIGMA2 = GPS_SIGMA2;
240  }
241 
242  float q[2][2];
243  q[0][0] = DT*DT*DT*DT/4.;
244  q[0][1] = DT*DT*DT/2.;
245  q[1][0] = DT*DT*DT/2.;
246  q[1][1] = DT*DT;
247 
248 
249  /* predict */
250  ins_alt += ins_alt_dot * DT;
251  p[0][0] = p[0][0]+p[1][0]*DT+DT*(p[0][1]+p[1][1]*DT) + SIGMA2*q[0][0];
252  p[0][1] = p[0][1]+p[1][1]*DT + SIGMA2*q[0][1];
253  p[1][0] = p[1][0]+p[1][1]*DT + SIGMA2*q[1][0];
254  p[1][1] = p[1][1] + SIGMA2*q[1][1];
255 
256  /* error estimate */
257  float e = p[0][0] + R;
258 
259  if (fabs(e) > 1e-5) {
260  float k_0 = p[0][0] / e;
261  float k_1 = p[1][0] / e;
262  e = z_meas - ins_alt;
263 
264  /* correction */
265  ins_alt += k_0 * e;
266  ins_alt_dot += k_1 * e;
267 
268  p[1][0] = -p[0][0]*k_1+p[1][0];
269  p[1][1] = -p[0][1]*k_1+p[1][1];
270  p[0][0] = p[0][0] * (1-k_0);
271  p[0][1] = p[0][1] * (1-k_0);
272  }
273 
274 #ifdef DEBUG_ALT_KALMAN
275  DOWNLINK_SEND_ALT_KALMAN(DefaultChannel,DefaultDevice,&(p[0][0]),&(p[0][1]),&(p[1][0]), &(p[1][1]));
276 #endif
277 }
