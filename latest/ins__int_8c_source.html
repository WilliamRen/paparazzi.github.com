187 void ins_propagate() {
188  /* untilt accels */
189  struct Int32Vect3 accel_meas_body;
190  INT32_RMAT_TRANSP_VMULT(accel_meas_body, imu.body_to_imu_rmat, imu.accel);
191  struct Int32Vect3 accel_meas_ltp;
192  INT32_RMAT_TRANSP_VMULT(accel_meas_ltp, (*stateGetNedToBodyRMat_i()), accel_meas_body);
193 
194  float z_accel_meas_float = ACCEL_FLOAT_OF_BFP(accel_meas_ltp.z);
195  if (ins_impl.baro_initialized) {
196  vff_propagate(z_accel_meas_float);
197  ins_impl.ltp_accel.z = ACCEL_BFP_OF_REAL(vff_zdotdot);
198  ins_impl.ltp_speed.z = SPEED_BFP_OF_REAL(vff_zdot);
199  ins_impl.ltp_pos.z = POS_BFP_OF_REAL(vff_z);
200  }
201  else { // feed accel from the sensors
202  // subtract -9.81m/s2 (acceleration measured due to gravity,
203  // but vehicle not accelerating in ltp)
204  ins_impl.ltp_accel.z = accel_meas_ltp.z + ACCEL_BFP_OF_REAL(9.81);
205  }
206 
207 #if USE_HFF
208  /* propagate horizontal filter */
209  b2_hff_propagate();
210  /* convert and copy result to ins_impl */
211  ins_update_from_hff();
212 #else
213  ins_impl.ltp_accel.x = accel_meas_ltp.x;
214  ins_impl.ltp_accel.y = accel_meas_ltp.y;
215 #endif /* USE_HFF */
216 
217  ins_ned_to_state();
218 }

220 static void baro_cb(uint8_t __attribute__((unused)) sender_id, const float *pressure) {
221  if (!ins_impl.baro_initialized) {
222  ins_impl.qfe = *pressure;
223  ins_impl.baro_initialized = TRUE;
224  }
225  if (ins.vf_realign) {
226  ins.vf_realign = FALSE;
227  ins_impl.qfe = *pressure;
228  vff_realign(0.);
229  ins_impl.ltp_accel.z = ACCEL_BFP_OF_REAL(vff_zdotdot);
230  ins_impl.ltp_speed.z = SPEED_BFP_OF_REAL(vff_zdot);
231  ins_impl.ltp_pos.z = POS_BFP_OF_REAL(vff_z);
232  }
233  else {
234  ins_impl.baro_z = -pprz_isa_height_of_pressure(*pressure, ins_impl.qfe);
235 #if USE_VFF_EXTENDED
236  vff_update_baro(ins_impl.baro_z);
237 #else
238  vff_update(ins_impl.baro_z);
239 #endif
240  }
241  ins_ned_to_state();
242 }

315 void ins_update_sonar() {
316 #if USE_SONAR
317  static float last_offset = 0.;
318  // new value filtered with median_filter
319  ins_impl.sonar_alt = update_median_filter(&ins_impl.sonar_median, sonar_meas);
320  float sonar = (ins_impl.sonar_alt - ins_impl.sonar_offset) * INS_SONAR_SENS;
321 
322 #ifdef INS_SONAR_VARIANCE_THRESHOLD
323  /* compute variance of error between sonar and baro alt */
324  float err = sonar + ins_impl.baro_z; // sonar positive up, baro positive down !!!!
325  var_err[var_idx] = err;
326  var_idx = (var_idx + 1) % VAR_ERR_MAX;
327  float var = variance_float(var_err, VAR_ERR_MAX);
328  DOWNLINK_SEND_INS_SONAR(DefaultChannel,DefaultDevice,&err, &sonar, &var);
329  //DOWNLINK_SEND_INS_SONAR(DefaultChannel,DefaultDevice,&ins_impl.sonar_alt, &sonar, &var);
330 #endif
331 
332  /* update filter assuming a flat ground */
333  if (sonar < INS_SONAR_MAX_RANGE
334 #ifdef INS_SONAR_MIN_RANGE
335  && sonar > INS_SONAR_MIN_RANGE
336 #endif
337 #ifdef INS_SONAR_THROTTLE_THRESHOLD
338  && stabilization_cmd[COMMAND_THRUST] < INS_SONAR_THROTTLE_THRESHOLD
339 #endif
340 #ifdef INS_SONAR_STAB_THRESHOLD
341  && stabilization_cmd[COMMAND_ROLL] < INS_SONAR_STAB_THRESHOLD
342  && stabilization_cmd[COMMAND_ROLL] > -INS_SONAR_STAB_THRESHOLD
343  && stabilization_cmd[COMMAND_PITCH] < INS_SONAR_STAB_THRESHOLD
344  && stabilization_cmd[COMMAND_PITCH] > -INS_SONAR_STAB_THRESHOLD
345  && stabilization_cmd[COMMAND_YAW] < INS_SONAR_STAB_THRESHOLD
346  && stabilization_cmd[COMMAND_YAW] > -INS_SONAR_STAB_THRESHOLD
347 #endif
348 #ifdef INS_SONAR_BARO_THRESHOLD
349  && ins_impl.baro_z > -INS_SONAR_BARO_THRESHOLD /* z down */
350 #endif
351 #ifdef INS_SONAR_VARIANCE_THRESHOLD
352  && var < INS_SONAR_VARIANCE_THRESHOLD
353 #endif
354  && ins_impl.update_on_agl
355  && ins_impl.baro_initialized) {
356  vff_update_alt_conf(-sonar, VFF_R_SONAR_0 + VFF_R_SONAR_OF_M * fabs(sonar));
357  last_offset = vff_offset;
358  }
359  else {
360  /* update offset with last value to avoid divergence */
361  vff_update_offset(last_offset);
362  }
363 #endif // USE_SONAR
364 }

383 }
