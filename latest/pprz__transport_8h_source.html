208 static inline void parse_pprz(struct pprz_transport * t, uint8_t c ) {
209  switch (t->status) {
210  case UNINIT:
211  if (c == STX)
212  t->status++;
213  break;
214  case GOT_STX:
215  if (t->trans.msg_received) {
216  t->trans.ovrn++;
217  goto error;
218  }
219  t->trans.payload_len = c-4; /* Counting STX, LENGTH and CRC1 and CRC2 */
220  t->ck_a = t->ck_b = c;
221  t->status++;
222  t->payload_idx = 0;
223  break;
224  case GOT_LENGTH:
225  t->trans.payload[t->payload_idx] = c;
226  t->ck_a += c; t->ck_b += t->ck_a;
227  t->payload_idx++;
228  if (t->payload_idx == t->trans.payload_len)
229  t->status++;
230  break;
231  case GOT_PAYLOAD:
232  if (c != t->ck_a)
233  goto error;
234  t->status++;
235  break;
236  case GOT_CRC1:
237  if (c != t->ck_b)
238  goto error;
239  t->trans.msg_received = TRUE;
240  goto restart;
241  default:
242  goto error;
243  }
244  return;
245  error:
246  t->trans.error++;
247  restart:
248  t->status = UNINIT;
249  return;
250 }
