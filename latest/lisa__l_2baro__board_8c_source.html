83 void baro_periodic(void) {
84  // check i2c_done
85  if (!i2c_idle(&i2c2)) return;
86 
87  switch (baro_board.status) {
88  case LBS_UNINITIALIZED:
89  baro_board_send_reset();
90  baro_board.status = LBS_RESETED;
91  break;
92  case LBS_RESETED:
93  baro_board_send_config_abs();
94  baro_board.status = LBS_INITIALIZING_ABS;
95  break;
96  case LBS_INITIALIZING_ABS:
97  baro_board_set_current_register(BARO_ABS_ADDR, 0x00);
98  baro_board.status = LBS_INITIALIZING_ABS_1;
99  break;
100  case LBS_INITIALIZING_ABS_1:
101  baro_board_send_config_diff();
102  baro_board.status = LBS_INITIALIZING_DIFF;
103  break;
104  case LBS_INITIALIZING_DIFF:
105  baro_board_set_current_register(BARO_DIFF_ADDR, 0x00);
106  baro_board.status = LBS_INITIALIZING_DIFF_1;
107  // baro_board.status = LBS_UNINITIALIZED;
108  break;
109  case LBS_INITIALIZING_DIFF_1:
110  baro_board.running = TRUE;
111  case LBS_READ_DIFF:
112  baro_board_read_from_current_register(BARO_ABS_ADDR);
113  baro_board.status = LBS_READING_ABS;
114  break;
115  case LBS_READ_ABS:
116  baro_board_read_from_current_register(BARO_DIFF_ADDR);
117  baro_board.status = LBS_READING_DIFF;
118  break;
119  default:
120  break;
121  }
122 
123 #ifdef BARO_LED
124  if (baro_board.running == TRUE) {
125  LED_ON(BARO_LED);
126  }
127  else {
128  LED_TOGGLE(BARO_LED);
129  }
130 #endif
131 }

133 void lisa_l_baro_event(void) {
134  if (baro_board.status == LBS_READING_ABS &&
135  baro_trans.status != I2CTransPending) {
136  baro_board.status = LBS_READ_ABS;
137  if (baro_trans.status == I2CTransSuccess) {
138  int16_t tmp = baro_trans.buf[0]<<8 | baro_trans.buf[1];
139  float pressure = LISA_L_BARO_SENS*(float)tmp;
140  AbiSendMsgBARO_ABS(BARO_BOARD_SENDER_ID, &pressure);
141  }
142  }
143  else if (baro_board.status == LBS_READING_DIFF &&
144  baro_trans.status != I2CTransPending) {
145  baro_board.status = LBS_READ_DIFF;
146  if (baro_trans.status == I2CTransSuccess) {
147  int16_t tmp = baro_trans.buf[0]<<8 | baro_trans.buf[1];
148  float diff = LISA_L_DIFF_SENS*(float)tmp;
149  AbiSendMsgBARO_DIFF(BARO_BOARD_SENDER_ID, &diff);
150  }
151  }
152 }

169 static inline void baro_board_send_reset(void) {
170  baro_trans.type = I2CTransTx;
171  baro_trans.slave_addr = 0x00;
172  baro_trans.len_w = 1;
173  baro_trans.buf[0] = 0x06;
174  i2c_submit(&i2c2,&baro_trans);
175 }
