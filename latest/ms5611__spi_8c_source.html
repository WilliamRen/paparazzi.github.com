90 void ms5611_spi_periodic_check(struct Ms5611_Spi *ms)
91 {
92  switch (ms->status) {
93  case MS5611_STATUS_RESET:
94  ms->status = MS5611_STATUS_RESET_OK;
95  break;
96  case MS5611_STATUS_RESET_OK:
97  if (ms->spi_trans.status == SPITransDone) {
98  /* start getting prom data */
99  ms->tx_buf[0] = MS5611_PROM_READ | (ms->prom_cnt << 1);
100  spi_submit(ms->spi_p, &(ms->spi_trans));
101  ms->status = MS5611_STATUS_PROM;
102  }
103  break;
104  case MS5611_STATUS_CONV_D1:
105  ms->status = MS5611_STATUS_CONV_D1_OK;
106  break;
107  case MS5611_STATUS_CONV_D1_OK:
108  if (ms->spi_trans.status == SPITransDone) {
109  /* read D1 adc */
110  ms->tx_buf[0] = MS5611_ADC_READ;
111  spi_submit(ms->spi_p, &(ms->spi_trans));
112  ms->status = MS5611_STATUS_ADC_D1;
113  }
114  break;
115  case MS5611_STATUS_CONV_D2:
116  ms->status = MS5611_STATUS_CONV_D2_OK;
117  break;
118  case MS5611_STATUS_CONV_D2_OK:
119  if (ms->spi_trans.status == SPITransDone) {
120  /* read D2 adc */
121  ms->tx_buf[0] = MS5611_ADC_READ;
122  spi_submit(ms->spi_p, &(ms->spi_trans));
123  ms->status = MS5611_STATUS_ADC_D2;
124  }
125  break;
126  default:
127  break;
128  }

131 void ms5611_spi_event(struct Ms5611_Spi *ms) {
132  if (ms->initialized) {
133  if (ms->spi_trans.status == SPITransFailed) {
134  ms->status = MS5611_STATUS_IDLE;
135  ms->spi_trans.status = SPITransDone;
136  }
137  else if (ms->spi_trans.status == SPITransSuccess) {
138  // Successfull reading
139  switch (ms->status) {
140 
141  case MS5611_STATUS_ADC_D1:
142  /* read D1 (pressure) */
143  ms->data.d1 = (ms->rx_buf[1] << 16) |
144  (ms->rx_buf[2] << 8) |
145  ms->rx_buf[3];
146  if (ms->data.d1 == 0) {
147  /* if value is zero, it was read to soon and is invalid, back to idle */
148  ms->status = MS5611_STATUS_IDLE;
149  }
150  else {
151  /* start D2 conversion */
152  ms->tx_buf[0] = MS5611_START_CONV_D2;
153  spi_submit(ms->spi_p, &(ms->spi_trans));
154  ms->status = MS5611_STATUS_CONV_D2;
155  }
156  break;
157 
158  case MS5611_STATUS_ADC_D2:
159  /* read D2 (temperature) */
160  ms->data.d2 = (ms->rx_buf[1] << 16) |
161  (ms->rx_buf[2] << 8) |
162  ms->rx_buf[3];
163  if (ms->data.d2 == 0) {
164  /* if value is zero, it was read to soon and is invalid, back to idle */
165  ms->status = MS5611_STATUS_IDLE;
166  }
167  else {
168  /* calculate temp and pressure from measurements */
169  ms5611_calc(&(ms->data));
170  ms->status = MS5611_STATUS_IDLE;
171  ms->data_available = TRUE;
172  }
173  break;
174 
175  default:
176  break;
177  }
178  ms->spi_trans.status = SPITransDone;
179  }
180  }
181  else if (ms->status != MS5611_STATUS_UNINIT) { // Configuring but not yet initialized
182  switch (ms->spi_trans.status) {
183 
184  case SPITransFailed:
185  /* try again */
186  ms->status = MS5611_STATUS_UNINIT;
187  ms->spi_trans.status = SPITransDone;
188  break;
189 
190  case SPITransSuccess:
191  if (ms->status == MS5611_STATUS_PROM) {
192  /* read prom data */
193  ms->data.c[ms->prom_cnt++] = (ms->rx_buf[1] << 8) |
194  ms->rx_buf[2];
195  if (ms->prom_cnt < PROM_NB) {
196  /* get next prom data */
197  ms->tx_buf[0] = MS5611_PROM_READ | (ms->prom_cnt << 1);
198  spi_submit(ms->spi_p, &(ms->spi_trans));
199  }
200  else {
201  /* done reading prom, check prom crc */
202  if (ms5611_prom_crc_ok(ms->data.c)) {
203  ms->initialized = TRUE;
204  ms->status = MS5611_STATUS_IDLE;
205  }
206  else {
207  /* checksum error, try again */
208  ms->status = MS5611_STATUS_UNINIT;
209  }
210  }
211  }
212  ms->spi_trans.status = SPITransDone;
213  break;
214 
215  default:
216  break;
217  }
218  }
