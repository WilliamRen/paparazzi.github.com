105 void electrical_periodic(void) {
106  static uint8_t bat_low_counter = 0;
107  static uint8_t bat_critical_counter = 0;
108 
109 #if defined(ADC_CHANNEL_VSUPPLY) && !defined(SITL)
110  electrical.vsupply = 10 * VoltageOfAdc((electrical_priv.vsupply_adc_buf.sum/electrical_priv.vsupply_adc_buf.av_nb_sample));
111 #endif
112 
113 #ifdef ADC_CHANNEL_CURRENT
114 #ifndef SITL
115  electrical.current = MilliAmpereOfAdc((electrical_priv.current_adc_buf.sum/electrical_priv.current_adc_buf.av_nb_sample));
116  /* Prevent an overflow on high current spikes when using the motor brake */
117  BoundAbs(electrical.current, 65000);
118 #endif
119 #elif defined MILLIAMP_AT_FULL_THROTTLE && defined COMMAND_CURRENT_ESTIMATION
120  /*
121  * Superellipse: abs(x/a)^n + abs(y/b)^n = 1
122  * with a = 1
123  * b = mA at full throttle
124  * n = 1.2 This defines nonlinearity (1 = linear)
125  * x = throttle
126  * y = current
127  *
128  * define CURRENT_ESTIMATION_NONLINEARITY in your airframe file to change the default nonlinearity factor of 1.2
129  */
130  float b = (float)MILLIAMP_AT_FULL_THROTTLE;
131  float x = ((float)commands[COMMAND_CURRENT_ESTIMATION]) / ((float)MAX_PPRZ);
132  /* electrical.current y = ( b^n - (b* x/a)^n )^1/n
133  * a=1, n = electrical_priv.nonlin_factor
134  */
135  electrical.current = b - pow((pow(b,electrical_priv.nonlin_factor)-pow((b*x),electrical_priv.nonlin_factor)), (1./electrical_priv.nonlin_factor));
136 #endif /* ADC_CHANNEL_CURRENT */
137 
138 
139  if (electrical.vsupply < LOW_BAT_LEVEL * 10) {
140  if (bat_low_counter > 0)
141  bat_low_counter--;
142  if (bat_low_counter == 0)
143  electrical.bat_low = TRUE;
144  }
145  else {
146  // reset battery low status and counter
147  bat_low_counter = BAT_CHECKER_DELAY * ELECTRICAL_PERIODIC_FREQ;
148  electrical.bat_low = FALSE;
149  }
150 
151  if (electrical.vsupply < CRITIC_BAT_LEVEL * 10) {

155  electrical.bat_critical = TRUE;
156  }
157  else {
158  // reset battery critical status and counter
159  bat_critical_counter = BAT_CHECKER_DELAY * ELECTRICAL_PERIODIC_FREQ;
160  electrical.bat_critical = FALSE;
161  }
