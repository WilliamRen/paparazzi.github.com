51 bool_t init_zamboni_survey(uint8_t center_wp, uint8_t dir_wp, float sweep_length, float sweep_spacing, int sweep_lines, float altitude)
52 {
53  zs.current_laps = 0;
54  zs.pre_leave_angle = 2;
55 
56  // copy variables from the flight plan
57  VECT2_COPY(zs.wp_center, waypoints[center_wp]);
58  VECT2_COPY(zs.wp_dir, waypoints[dir_wp]);
59  zs.altitude = altitude;
60 
61  // if turning right leave circle before angle is reached, if turning left - leave after
62  if (sweep_spacing > 0) {
63  zs.pre_leave_angle -= zs.pre_leave_angle;
64  }
65 
66  struct FloatVect2 flight_vec;
67  VECT2_DIFF(flight_vec, zs.wp_dir, zs.wp_center);
68  FLOAT_VECT2_NORMALIZE(flight_vec);
69 
70  // calculate the flight_angle
71  zs.flight_angle = DegOfRad(atan2(flight_vec.x, flight_vec.y));
72  zs.return_angle = zs.flight_angle + 180;
73  if (zs.return_angle > 359) {
74  zs.return_angle -= 360;
75  }
76 
77  // calculate the vector from one flightline perpendicular to the next flightline left,
78  // seen in the flightdirection. (Delta x and delta y betwen two adjecent flightlines)
79  // (used later to move the flight pattern one flightline up for each round)
80  zs.sweep_width.x = -flight_vec.y * sweep_spacing;
81  zs.sweep_width.y = +flight_vec.x * sweep_spacing;
82 
83  // calculate number of laps to fly and turning radius for each end
84  zs.total_laps = (sweep_lines+1)/2;
85  zs.turnradius1 = (zs.total_laps-1) * sweep_spacing * 0.5;
86  zs.turnradius2 = zs.total_laps * sweep_spacing * 0.5;
87 
88  struct FloatVect2 flight_line;
89  VECT2_SMUL(flight_line, flight_vec, sweep_length * 0.5);
90 
91  //CALCULATE THE NAVIGATION POINTS
92  //start and end of flight-line in flight-direction
93  VECT2_DIFF(zs.seg_start, zs.wp_center, flight_line);
94  VECT2_SUM(zs.seg_end, zs.wp_center, flight_line);
95 
96  struct FloatVect2 sweep_span;
97  VECT2_SMUL(sweep_span, zs.sweep_width, zs.total_laps-1);
98  //start and end of flight-line in return-direction
99  VECT2_DIFF(zs.ret_start, zs.seg_end, sweep_span);
100  VECT2_DIFF(zs.ret_end, zs.seg_start, sweep_span);
101 
102  //turn-centers at both ends
103  zs.turn_center1.x = (zs.seg_end.x + zs.ret_start.x) / 2.0;
104  zs.turn_center1.y = (zs.seg_end.y + zs.ret_start.y) / 2.0;
105  zs.turn_center2.x = (zs.seg_start.x + zs.ret_end.x + zs.sweep_width.x) / 2.0;
106  zs.turn_center2.y = (zs.seg_start.y + zs.ret_end.y + zs.sweep_width.y) / 2.0;
107 
108  //fast climbing to desired altitude
109  NavVerticalAutoThrottleMode(100.0);
110  NavVerticalAltitudeMode(zs.altitude, 0.0);
111 
112  zs.stage = Z_ENTRY;
113 
114  return FALSE;
115 }

124 bool_t zamboni_survey(void)
125 {
126  // retain altitude
127  NavVerticalAutoThrottleMode(0.0);
128  NavVerticalAltitudeMode(zs.altitude, 0.0);
129 
130  //go from center of field to end of field - (before starting the syrvey)
131  if (zs.stage == Z_ENTRY) {
132  nav_route_xy(zs.wp_center.x, zs.wp_center.y, zs.seg_end.x, zs.seg_end.y);
133  if (nav_approaching_xy(zs.seg_end.x, zs.seg_end.y, zs.wp_center.x, zs.wp_center.y, CARROT)) {
134  zs.stage = Z_TURN1;
135  NavVerticalAutoThrottleMode(0.0);
136  nav_init_stage();
137  }
138  }
139 
140  //Turn from stage to return
141  else if (zs.stage == Z_TURN1) {
142  nav_circle_XY(zs.turn_center1.x, zs.turn_center1.y, zs.turnradius1);
143  if (NavCourseCloseTo(zs.return_angle + zs.pre_leave_angle)){
144  // && nav_approaching_xy(zs.seg_end.x, zs.seg_end.y, zs.seg_start.x, zs.seg_start.y, CARROT
145  //calculate SEG-points for the next flyover
146  VECT2_ADD(zs.seg_start, zs.sweep_width);
147  VECT2_ADD(zs.seg_end, zs.sweep_width);
148 
149  zs.stage = Z_RET;
150  nav_init_stage();
151 #ifdef DIGITAL_CAM
152  LINE_START_FUNCTION;
153 #endif
154  }
155  }
156 
157  //fly the segment until seg_end is reached
158  else if (zs.stage == Z_RET) {
159  nav_route_xy(zs.ret_start.x, zs.ret_start.y, zs.ret_end.x, zs.ret_end.y);
160  if (nav_approaching_xy(zs.ret_end.x, zs.ret_end.y, zs.ret_start.x, zs.ret_start.y, 0)) {
161  zs.current_laps = zs.current_laps + 1;
162 #ifdef DIGITAL_CAM
163  //dc_stop();
164  LINE_STOP_FUNCTION;
165 #endif
166  zs.stage = Z_TURN2;
167  }
168  }
169 
170  //turn from stage to return
171  else if (zs.stage == Z_TURN2) {
172  nav_circle_XY(zs.turn_center2.x, zs.turn_center2.y, zs.turnradius2);
173  if (NavCourseCloseTo(zs.flight_angle + zs.pre_leave_angle)) {
174  //zs.current_laps = zs.current_laps + 1;
175  zs.stage = Z_SEG;
176  nav_init_stage();
177 #ifdef DIGITAL_CAM
178  LINE_START_FUNCTION;
179 #endif
180  }
181 
182  //return
183  } else if (zs.stage == Z_SEG) {
184  nav_route_xy(zs.seg_start.x, zs.seg_start.y, zs.seg_end.x, zs.seg_end.y);
185  if (nav_approaching_xy(zs.seg_end.x, zs.seg_end.y, zs.seg_start.x, zs.seg_start.y, 0)) {
186 
187  // calculate the rest of the points for the next fly-over
188  VECT2_ADD(zs.ret_start, zs.sweep_width);
189  VECT2_ADD(zs.ret_end, zs.sweep_width);
190  zs.turn_center1.x = (zs.seg_end.x + zs.ret_start.x)/2;
191  zs.turn_center1.y = (zs.seg_end.y + zs.ret_start.y)/2;
192  zs.turn_center2.x = (zs.seg_start.x + zs.ret_end.x + zs.sweep_width.x) / 2;
193  zs.turn_center2.y = (zs.seg_start.y + zs.ret_end.y + zs.sweep_width.y) / 2;
194 
195  zs.stage = Z_TURN1;
196  nav_init_stage();
197 #ifdef DIGITAL_CAM
198  //dc_stop();
199  LINE_STOP_FUNCTION;
200 #endif
201  }
202  }
203  if (zs.current_laps >= zs.total_laps) {
204 #ifdef DIGITAL_CAM
205  LINE_STOP_FUNCTION;
206 #endif
207  return FALSE;
208  }
209  else {
210  return TRUE;
211  }
212 }
