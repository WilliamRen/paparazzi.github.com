40 static void send_gps(void) {
41  static uint8_t i;
42  int16_t climb = -gps.ned_vel.z;
43  int16_t course = (DegOfRad(gps.course)/((int32_t)1e6));
44  DOWNLINK_SEND_GPS(DefaultChannel, DefaultDevice, &gps.fix,
45  &gps.utm_pos.east, &gps.utm_pos.north,
46  &course, &gps.hmsl, &gps.gspeed, &climb,
47  &gps.week, &gps.tow, &gps.utm_pos.zone, &i);
48  if ((gps.fix != GPS_FIX_3D) && (i >= gps.nb_channels)) i = 0;
49  if (i >= gps.nb_channels * 2) i = 0;
50  if (i < gps.nb_channels && ((gps.fix != GPS_FIX_3D) || (gps.svinfos[i].cno > 0))) {
51  DOWNLINK_SEND_SVINFO(DefaultChannel, DefaultDevice, &i,
52  &gps.svinfos[i].svid, &gps.svinfos[i].flags,
53  &gps.svinfos[i].qi, &gps.svinfos[i].cno,
54  &gps.svinfos[i].elev, &gps.svinfos[i].azim);
55  }
56  i++;
57 }

78  last_cnos[i] = gps.svinfos[i].cno;
79  }
80  i++;
81 }

117 uint32_t gps_tow_from_sys_ticks(uint32_t sys_ticks)
118 {
119  uint32_t clock_delta;
120  uint32_t time_delta;
121  uint32_t itow_now;
122 
123  if (sys_ticks < gps_time_sync.t0_ticks) {
124  clock_delta = (0xFFFFFFFF - sys_ticks) + gps_time_sync.t0_ticks + 1;
125  } else {
126  clock_delta = sys_ticks - gps_time_sync.t0_ticks;
127  }
128 
129  time_delta = msec_of_sys_time_ticks(clock_delta);
130 
131  itow_now = gps_time_sync.t0_tow + time_delta;
132  if (itow_now > MSEC_PER_WEEK) {
133  itow_now %= MSEC_PER_WEEK;
134  }
135 
136  return itow_now;
137 }
