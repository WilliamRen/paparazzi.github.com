102 static void decode_sbus_buffer (const uint8_t *src, uint16_t *dst, bool_t *available)
103 {
104  // reset counters
105  uint8_t byteInRawBuf = 0;
106  uint8_t bitInRawBuf = 0;
107  uint8_t channel = 0;
108  uint8_t bitInChannel = 0;
109 
110  // clear bits
111  memset (dst, 0, SBUS_NB_CHANNEL*sizeof(uint16_t));
112 
113  // decode sbus data
114  for (uint8_t i=0; i< (SBUS_NB_CHANNEL*SBUS_BIT_PER_CHANNEL); i++) {
115  if (src[byteInRawBuf] & (1<<bitInRawBuf))
116  dst[channel] |= (1<<bitInChannel);
117 
118  bitInRawBuf++;
119  bitInChannel++;
120 
121  if (bitInRawBuf == SBUS_BIT_PER_BYTE) {
122  bitInRawBuf = 0;
123  byteInRawBuf++;
124  }
125  if (bitInChannel == SBUS_BIT_PER_CHANNEL) {
126  bitInChannel = 0;
127  channel++;
128  }
129  }
130  // test frame lost flag
131  *available = !bit_is_set(src[SBUS_FLAGS_BYTE], SBUS_FRAME_LOST_BIT);
132 }

134 // Decoding event function
135 // Reading from UART
136 void sbus_decode_event(void) {
137  uint8_t rbyte;
138  if (uart_char_available(&SBUS_UART_DEV)) {
139  do {
140  rbyte = uart_getch(&SBUS_UART_DEV);
141  switch (sbus.status) {
142  case SBUS_STATUS_UNINIT:
143  // Wait for the start byte
144  if (rbyte == SBUS_START_BYTE) {
145  sbus.status++;
146  sbus.idx = 0;
147  }
148  break;
149  case SBUS_STATUS_GOT_START:
150  // Store buffer
151  sbus.buffer[sbus.idx] = rbyte;
152  sbus.idx++;
153  if (sbus.idx == SBUS_BUF_LENGTH) {
154  // Decode if last byte is the correct end byte
155  if (rbyte == SBUS_END_BYTE) {
156  decode_sbus_buffer(sbus.buffer, sbus.pulses, &sbus.frame_available);
157  }
158  sbus.status = SBUS_STATUS_UNINIT;
159  }
160  break;
161  default:
162  break;
163  }
164  } while (uart_char_available(&SBUS_UART_DEV));
165  }
