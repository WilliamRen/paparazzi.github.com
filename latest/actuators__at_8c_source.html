36 void actuators_set(pprz_t commands[]) {
37  //Calculate the thrus, roll, pitch and yaw from the PPRZ commands
38  float thrust = ((float)(commands[COMMAND_THRUST]-MAX_PPRZ/2) / (float)MAX_PPRZ)*2.0f;
39  float roll = ((float)commands[COMMAND_ROLL] / (float)MAX_PPRZ);
40  float pitch = ((float)commands[COMMAND_PITCH] / (float)MAX_PPRZ);
41  float yaw = ((float)commands[COMMAND_YAW] / (float)MAX_PPRZ);
42 
43  //Starting engine
44  if(thrust > 0 && (ahrs_impl.control_state == CTRL_DEFAULT || ahrs_impl.control_state == CTRL_INIT || ahrs_impl.control_state == CTRL_LANDED))
45  at_com_send_ref(REF_TAKEOFF);
46 
47  //Check emergency or stop engine
48  if((ahrs_impl.state & ARDRONE_EMERGENCY_MASK) != 0)
49  at_com_send_ref(REF_EMERGENCY);
50  else if(thrust < -0.9 && !(ahrs_impl.control_state == CTRL_DEFAULT || ahrs_impl.control_state == CTRL_INIT || ahrs_impl.control_state == CTRL_LANDED))
51  at_com_send_ref(0);
52 
53  //Calibration
54  if((ahrs_impl.state & ARDRONE_MAGNETO_NEEDS_CALIB) != 0 && (ahrs_impl.control_state == CTRL_FLYING || ahrs_impl.control_state == CTRL_HOVERING))
55  at_com_send_calib(0);
56 
57  //Moving
58  if((ahrs_impl.state & ARDRONE_MAGNETO_NEEDS_CALIB) == 0 && (ahrs_impl.control_state == CTRL_FLYING || ahrs_impl.control_state == CTRL_HOVERING))
59  at_com_send_pcmd(1, thrust, roll, pitch, yaw);
60 
61  //Keep alive (FIXME)
62  at_com_send_config("general:navdata_demo", "FALSE");
63 }
