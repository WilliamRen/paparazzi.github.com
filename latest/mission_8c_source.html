44 bool_t mission_insert(enum MissionInsertMode insert, struct _mission_element * element) {
45  uint8_t tmp;
46 
47  switch (insert) {
48  case Append:
49  tmp = (mission.insert_idx + 1) % MISSION_ELEMENT_NB;
50  if (tmp == mission.current_idx) return FALSE; // no room to insert element
51  memcpy(&mission.elements[mission.insert_idx], element, sizeof(struct _mission_element)); // add element
52  mission.insert_idx = tmp; // move insert index
53  break;
54  case Prepend:
55  if (mission.current_idx == 0) tmp = MISSION_ELEMENT_NB-1;
56  else tmp = mission.current_idx - 1;
57  if (tmp == mission.insert_idx) return FALSE; // no room to inser element
58  memcpy(&mission.elements[tmp], element, sizeof(struct _mission_element)); // add element
59  mission.current_idx = tmp; // move current index
60  break;
61  case ReplaceCurrent:
62  // current element can always be modified, index are not changed
63  memcpy(&mission.elements[mission.current_idx], element, sizeof(struct _mission_element));
64  break;
65  case ReplaceAll:
66  // reset queue and index
67  memcpy(&mission.elements[0], element, sizeof(struct _mission_element));
68  mission.insert_idx = 0;
69  mission.current_idx = 0;
70  break;
71  default:
72  // unknown insertion mode
73  return FALSE;
74  }
75  return TRUE;
76 
77 }
