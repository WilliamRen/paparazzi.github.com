57 void baro_periodic( void ) {
58 
59  // Run some loops to get correct readings from the adc
60  if (startup_cnt > 0) {
61  --startup_cnt;
62 #ifdef BARO_LED
63  LED_TOGGLE(BARO_LED);
64  if (startup_cnt == 0) {
65  LED_ON(BARO_LED);
66  }
67 #endif
68  }
69  // Read the ADC
70  ads1114_read(&BARO_ABS_ADS);
71 }

73 void umarim_baro_event(void) {
