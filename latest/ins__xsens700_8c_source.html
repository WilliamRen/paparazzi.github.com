270 void handle_ins_msg( void) {
271 
272 
273  // Send to Estimator (Control)
274 #ifdef XSENS_BACKWARDS
275  EstimatorSetAtt((-ins_phi+ins_roll_neutral), (RadOfDeg(-90)-ins_psi), (ins_theta+ins_pitch_neutral));
276  EstimatorSetRate(ins_p,-ins_q, -ins_r);
277 #else
278  EstimatorSetAtt(ins_phi+ins_roll_neutral,RadOfDeg(180) -ins_psi, -ins_theta+ins_pitch_neutral);
279  EstimatorSetRate(ins_p, ins_q, ins_r);
280 #endif
281 
282  // Position
283  float gps_east = gps.utm_pos.east / 100.;
284  float gps_north = gps.utm_pos.north / 100.;
285  gps_east -= nav_utm_east0;
286  gps_north -= nav_utm_north0;
287  EstimatorSetPosXY(gps_east, gps_north);
288 
289  // Altitude and vertical speed
290  float hmsl = gps.hmsl;
291  hmsl /= 1000.0f;
292  EstimatorSetAlt(hmsl);
293 
294  #ifndef ALT_KALMAN
295  #warning NO_VZ
296  #endif
297 
298  // Horizontal speed
299  float fspeed = sqrt(ins_vx*ins_vx + ins_vy*ins_vy);
300  if (gps.fix != GPS_FIX_3D)
301  {
302  fspeed = 0;
303  }
304  float fclimb = -ins_vz;
305  float fcourse = atan2f((float)ins_vx, (float)ins_vy);
306  EstimatorSetSpeedPol(fspeed, fcourse, fclimb);
307 
308  // Now also finish filling the gps struct for telemetry purposes
309  gps.gspeed = (fspeed * 100.);
310  gps.speed_3d = (uint16_t)(sqrt(ins_vx*ins_vx + ins_vy*ins_vy + ins_vz*ins_vz) * 100.);
311  gps.course = fcourse * 1e7;
312 
313 }

315 void parse_ins_msg( void ) {
316  uint8_t offset = 0;
317  if (xsens_id == XSENS_ReqLeverArmGpsAck_ID) {
318  xsens_gps_arm_x = XSENS_ReqLeverArmGpsAck_x(xsens_msg_buf);
319  xsens_gps_arm_y = XSENS_ReqLeverArmGpsAck_y(xsens_msg_buf);
320  xsens_gps_arm_z = XSENS_ReqLeverArmGpsAck_z(xsens_msg_buf);
321 
322  DOWNLINK_SEND_IMU_MAG_SETTINGS(DefaultChannel, DefaultDevice,&xsens_declination,&xsens_declination,&xsens_gps_arm_x,&xsens_gps_arm_y,&xsens_gps_arm_z);
323  }
324  else if (xsens_id == XSENS_Error_ID) {
325  xsens_errorcode = XSENS_Error_errorcode(xsens_msg_buf);
326  }
327  else if (xsens_id == XSENS_MTData2_ID) {
328  for (offset=0;offset<xsens_len;) {
329  uint8_t code1 = xsens_msg_buf[offset];
330  uint8_t code2 = xsens_msg_buf[offset+1];
331  int subpacklen = (int)xsens_msg_buf[offset+2];
332  offset += 3;
333 
334 
335  if (code1 == 0x10)
336  {
337  if (code2 == 0x10)
338  {
339  // UTC
340  }
341  else if (code2 == 0x20)
342  {
343  // Packet Counter
344  }
345  if (code2 == 0x30)
346  {
347  // ITOW
348  }
349  }
350  else if (code1 == 0x20)
351  {
352  if (code2 == 0x30)
353  {
354  // Attitude Euler
355  ins_phi = XSENS_DATA_Euler_roll(xsens_msg_buf,offset) * M_PI / 180;
356  ins_theta = XSENS_DATA_Euler_pitch(xsens_msg_buf,offset) * M_PI / 180;
357  ins_psi = XSENS_DATA_Euler_yaw(xsens_msg_buf,offset) * M_PI / 180;
358 
359  new_ins_attitude = 1;
360 
361  }
362  }
363  else if (code1 == 0x40)
364  {
365  if (code2 == 0x10)
366  {
367  // Delta-V
368  ins_ax = XSENS_DATA_Acceleration_x(xsens_msg_buf,offset)*100.0f;
369  ins_ay = XSENS_DATA_Acceleration_y(xsens_msg_buf,offset)*100.0f;
370  ins_az = XSENS_DATA_Acceleration_z(xsens_msg_buf,offset)*100.0f;
371  }
372  }
373  else if (code1 == 0x80)
374  {
375  if (code2 == 0x20)
376  {
377  // Rate Of Turn
378  ins_p = XSENS_DATA_RateOfTurn_x(xsens_msg_buf,offset) * M_PI / 180;
379  ins_q = XSENS_DATA_RateOfTurn_y(xsens_msg_buf,offset) * M_PI / 180;
380  ins_r = XSENS_DATA_RateOfTurn_z(xsens_msg_buf,offset) * M_PI / 180;
381  }
382  }
383  else if (code1 == 0x30)
384  {
385  if (code2 == 0x10)
386  {
387  // Baro Pressure
388  }
389  }
390  else if (code1 == 0xE0)
391  {
392  if (code2 == 0x20)
393  {
394  // Status Word
395  xsens_msg_statusword = XSENS_DATA_StatusWord_status(xsens_msg_buf,offset);
396  //gps.tow = xsens_msg_statusword;
397 #if USE_GPS_XSENS
398  if (bit_is_set(xsens_msg_statusword,2) && bit_is_set(xsens_msg_statusword,1))
399  {
400  if (bit_is_set(xsens_msg_statusword,23) && bit_is_set(xsens_msg_statusword,24))
401  gps.fix = GPS_FIX_3D;
402  else
403  gps.fix = GPS_FIX_2D;
404  }
405  else gps.fix = GPS_FIX_NONE;
406 #endif
407  }
408  }
409  else if (code1 == 0x88)
410  {
411  if (code2 == 0x40)
412  {
413  gps.week = XSENS_DATA_GpsSol_week(xsens_msg_buf,offset);
414  gps.num_sv = XSENS_DATA_GpsSol_numSv(xsens_msg_buf,offset);
415  gps.pacc = XSENS_DATA_GpsSol_pAcc(xsens_msg_buf,offset);
416  gps.sacc = XSENS_DATA_GpsSol_sAcc(xsens_msg_buf,offset);
417  gps.pdop = XSENS_DATA_GpsSol_pDop(xsens_msg_buf,offset);
418  }
419  else if (code2 == 0xA0)
420  {
421  // SVINFO
422  gps.tow = XSENS_XDI_GpsSvInfo_iTOW(xsens_msg_buf+offset);
423 
424 #if USE_GPS_XSENS

426 
427  gps.last_fix_time = sys_time.nb_sec;

439 #endif
440  }
441  }

443  {

445  {

452

454  float geoid_h;
455  WGS84_ELLIPSOID_TO_GEOID(lla_f.lat,lla_f.lon,geoid_h);
456  gps.hmsl = gps.utm_pos.alt - (geoid_h * 1000.0f);
457 
458  //gps.tow = geoid_h * 1000.0f; //gps.utm_pos.alt;
459  }
460  else if (code2 == 0x40)
461  {
462  // LatLong
463 #ifdef GPS_LED
464  LED_TOGGLE(GPS_LED);
465 #endif
466  gps.last_fix_time = sys_time.nb_sec;
467  gps.week = 0; // FIXME

469  lla_f.lon = RadOfDeg(XSENS_DATA_LatLon_lon(xsens_msg_buf,offset));

481  }

494  }

497  subpacklen = 0;
498  offset += subpacklen;
499  }

532 void parse_ins_buffer( uint8_t c ) {
533  ck += c;
534  switch (xsens_status) {
535  case UNINIT:
536  if (c != XSENS_START)
537  goto error;
538  xsens_status++;
539  ck = 0;
540  break;
541  case GOT_START:
542  if (c != XSENS_BID)
543  goto error;
544  xsens_status++;
545  break;
546  case GOT_BID:
547  xsens_id = c;
548  xsens_status++;
549  break;
550  case GOT_MID:
551  xsens_len = c;
552  if (xsens_len > XSENS_MAX_PAYLOAD)
553  goto error;
554  xsens_msg_idx = 0;
555  xsens_status++;
556  break;
557  case GOT_LEN:
558  xsens_msg_buf[xsens_msg_idx] = c;
559  xsens_msg_idx++;
560  if (xsens_msg_idx >= xsens_len)
561  xsens_status++;
562  break;
563  case GOT_DATA:
564  if (ck != 0)
565  goto error;
566  ins_msg_received = TRUE;
567  goto restart;
568  break;
569  }
570  return;
571  error:
572  restart:
573  xsens_status = UNINIT;
574  return;
575 }
