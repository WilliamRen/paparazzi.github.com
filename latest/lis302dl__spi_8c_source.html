72 // Configuration function called once before normal use
73 static void lis302dl_spi_send_config(struct Lis302dl_Spi *lis)
74 {
75  uint8_t reg_val = 0;
76 
77  switch (lis->init_status) {
78  case LIS_CONF_WHO_AM_I:
79  /* query device id */
80  lis->spi_trans.output_length = 1;
81  lis->spi_trans.input_length = 2;
82  /* set read bit then reg address */
83  lis->tx_buf[0] = (1<<7 | LIS302DL_REG_WHO_AM_I);
84  if (spi_submit(lis->spi_p, &(lis->spi_trans)))
85  lis->init_status++;
86  break;
87  case LIS_CONF_REG2:
88  /* set SPI mode, Filtered Data Selection */
89  reg_val = (lis->config.spi_3_wire << 7) | (lis->config.filt_data << 4);
90  lis302dl_spi_write_to_reg(lis, LIS302DL_REG_CTRL_REG2, reg_val);
91  lis->init_status++;
92  break;
93  case LIS_CONF_REG3:
94  /* Interrupt active high/low */
95  lis302dl_spi_write_to_reg(lis, LIS302DL_REG_CTRL_REG3, (lis->config.int_invert << 7));
96  lis->init_status++;
97  break;
98  case LIS_CONF_ENABLE:
99  /* set data rate, range, enable measurement, is in standby after power up */
100  reg_val = (lis->config.rate << 7) |
101  (1 << 6) | // Power Down Control to active mode
102  (lis->config.range << 5) |
103  0x5; // enable z,y,x axes
104  lis302dl_spi_write_to_reg(lis, LIS302DL_REG_CTRL_REG1, reg_val);
105  lis->init_status++;
106  break;
107  case LIS_CONF_DONE:
108  lis->initialized = TRUE;
109  lis->spi_trans.status = SPITransDone;
110  break;
111  default:
112  break;
113  }
114 }

126 void lis302dl_spi_read(struct Lis302dl_Spi *lis)
127 {
128  if (lis->initialized && lis->spi_trans.status == SPITransDone) {
129  lis->spi_trans.output_length = 1;
130  lis->spi_trans.input_length = 8;
131  /* set read bit and multiple byte bit, then address */
132  lis->tx_buf[0] = (1<<7|1<<6|LIS302DL_REG_STATUS);
133  spi_submit(lis->spi_p, &(lis->spi_trans));
134  }

137 void lis302dl_spi_event(struct Lis302dl_Spi *lis)
138 {
139  if (lis->initialized) {
140  if (lis->spi_trans.status == SPITransFailed) {
141  lis->spi_trans.status = SPITransDone;
142  }
143  else if (lis->spi_trans.status == SPITransSuccess) {
144  // Successfull reading
145  if (bit_is_set(lis->rx_buf[1], 3)) {
146  // new xyz data available
147  lis->data.vect.x = lis->rx_buf[3];
148  lis->data.vect.y = lis->rx_buf[5];
149  lis->data.vect.z = lis->rx_buf[7];
150  lis->data_available = TRUE;
151  }
152  lis->spi_trans.status = SPITransDone;
153  }
154  }
155  else if (lis->init_status != LIS_CONF_UNINIT) { // Configuring but not yet initialized
156  switch (lis->spi_trans.status) {
157  case SPITransFailed:
158  lis->init_status--; // Retry config (TODO max retry)
159  case SPITransSuccess:
160  if (lis->init_status == LIS_CONF_WHO_AM_I_OK) {
161  if (lis->rx_buf[1] == LIS302DL_WHO_AM_I)
162  lis->init_status++;
163  else
164  lis->init_status = LIS_CONF_WHO_AM_I;
165  }
166  case SPITransDone:
167  lis->spi_trans.status = SPITransDone;
168  lis302dl_spi_send_config(lis);
169  break;
170  default:
171  break;
172  }
173  }
