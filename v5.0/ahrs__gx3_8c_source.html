64 /* Big Endian to Float */
65 static inline float bef(volatile uint8_t *c) {
66  float f;
67  int8_t * p;
68  p = ((int8_t *)&f)+3;
69  *p-- = *c++;
70  *p-- = *c++;
71  *p-- = *c++;
72  *p = *c;
73  return f;
74 }

76 static inline bool_t GX3_verify_chk(volatile uint8_t *buff_add) {
77  uint16_t i,chk_calc;
78  chk_calc = 0;
79  for (i=0;i<GX3_MSG_LEN-2;i++) {
80  chk_calc += (uint8_t)*buff_add++;
81  }
82  return (chk_calc == ( (((uint16_t)*buff_add)<<8) + (uint8_t)*(buff_add+1) ));
83 }

241 /* GX3 Packet Collection */
242 void GX3_packet_parse( uint8_t c ) {
243  switch (GX3_packet.status) {
244  case GX3PacketWaiting:
245  GX3_packet.msg_idx = 0;
246  if (c == GX3_HEADER) {
247  GX3_packet.status++;
248  GX3_packet.msg_buf[GX3_packet.msg_idx] = c;
249  GX3_packet.msg_idx++;
250  } else {
251  GX3_packet.hdr_error++;
252  }
253  break;
254  case GX3PacketReading:
255  GX3_packet.msg_buf[GX3_packet.msg_idx] = c;
256  GX3_packet.msg_idx++;
257  if (GX3_packet.msg_idx == GX3_MSG_LEN) {
258  if (GX3_verify_chk(GX3_packet.msg_buf)) {
259  GX3_packet.msg_available = TRUE;
260  } else {
261  GX3_packet.msg_available = FALSE;
262  GX3_packet.chksm_error++;
263  }
264  GX3_packet.status = 0;
265  }
266  break;
267  default:
268  GX3_packet.status = 0;
269  GX3_packet.msg_idx = 0;
270  break;
271  }
272 }
